455 sf::Vector2f Game::tileCollision(const int levelTiles[], int nonColliderTile, int tileNumber, sf::Vector2i gridPosition, sf::Vector2f previousPos, sf::FloatRect rectBounds) {

480 if (rectBounds.intersects(tileBounds))

483 }
484 }

488 void Game::update() {
489 updateSFMLEvents();
490 view = window.getView();
491 currentHeroPos = currentHero->getPos();
492
493 // Collision management
494 bool collisionEnabled = true;
495 playerBounds = sf::FloatRect(currentHeroPos.x, currentHeroPos.y, 16*4,16*4);
496 sf::Vector2i playerGridPosition = sf::Vector2i(0,0);
497 playerGridPosition.x = round(playerBounds.left / (16*4));
498 playerGridPosition.y = round(playerBounds.top / (16*4));
499 int tileNumber = (playerGridPosition.y * mapSize.x + playerGridPosition.x);
500 // Bridge river collision removal
501 sf::FloatRect bridgeBounds = sf::FloatRect(14*16*4, 14*16*4, 16*4, 16*4*5);
502 sf::FloatRect intersect;
503 if (playerBounds.intersects(bridgeBounds) && enableBridge) {
504 collisionEnabled = false;
505 if (playerBounds.intersects(bridgeBounds, intersect)) {
506 if (intersect.width < intersect.height) {
507 currentHeroPos.x = bridgeBounds.left;
508 }
509 }
510 }
511 // Tunnel cliff collision removal
512 if (!onCliff) {
513 for(const sf::FloatRect bounds : tunnelBounds) {
514 if (playerBounds.intersects(bounds)) {
515 collisionEnabled = false;
516 if (playerBounds.intersects(bounds, intersect)) {
517 if (playerGridPosition.x < 10 || (playerGridPosition.x < 14 && playerGridPosition.x > 10)) {
518 if (intersect.height < intersect.width) {
519 currentHeroPos.y = bounds.top;
520 }
521 }
522 else if (playerGridPosition == sf::Vector2i(10,6) || playerGridPosition == sf::Vector2i(14,2)) {
523 if (playerBounds.left+playerBounds.width > bounds.left + bounds.width) {
524 currentHeroPos.x = bounds.left+bounds.width-playerBounds.width;
525 }
526 if (playerBounds.top + playerBounds.height > bounds.top + bounds.height) {
527 currentHeroPos.y = bounds.top+bounds.height-playerBounds.height;
528 }
529
530 }
531 else if (playerGridPosition == sf::Vector2i(10,2)) {
532 if (playerBounds.left < bounds.left) {
533 currentHeroPos.x = bounds.left;
534 }
535 if (playerBounds.top < bounds.top) {
536 currentHeroPos.y = bounds.top;
537 }
538
539 }
540 else if (playerGridPosition.x == 10 || playerGridPosition.x == 14) {
541 if (intersect.width < intersect.height) {
542 currentHeroPos.x = bounds.left;
543 }
544 }
545 }
546 }
547 }
548 }
549 else { // On cliff scene
550 collisionEnabled = false;
551 currentHeroPos = tileCollision(levelCliff, 15, tileNumber, playerGridPosition, currentHeroPos, playerBounds);
552 if (chest.getCollision()) {
553 currentHeroPos = chest.tileCollision(playerGridPosition, currentHeroPos, playerBounds);
554 }
555 }
556 // Default scene
557 if (collisionEnabled) {
558 currentHeroPos = tileCollision(levelWater, -1, tileNumber, playerGridPosition, currentHeroPos, playerBounds);
559 currentHeroPos = tileCollision(levelCliff, -1, tileNumber, playerGridPosition, currentHeroPos, playerBounds);
560 currentHeroPos = tileCollision(stoneWallPlatform, -1, tileNumber, playerGridPosition, currentHeroPos, playerBounds);
561 currentHeroPos = tileCollision(stonePlatformSequence, -1, tileNumber, playerGridPosition, currentHeroPos, playerBounds);
562
563 for (Interactable *interactable : interactables) {
564 if (interactable->getCollision()) {
565 currentHeroPos = interactable->tileCollision(playerGridPosition, currentHeroPos, playerBounds);
566 }
567 }
568 }
569
570 // Checks for collision on left side of map
571 if (currentHero->getPos().x < 0.f){
572 currentHeroPos.x = 0.f;
573 }
574 // Checks for collision on right side of map
575 if ((currentHero->getPos().x+currentHero->getFrameSize()+currentHero->getFrameSize()/2) > mapSize.x*16*4) {
576 currentHeroPos.x = mapSize.x*16*4-currentHero->getFrameSize()-currentHero->getFrameSize()/2;
577 }
578 // Checks for collision on top of map
579 if (currentHero->getPos().y < 0.f) {
580 currentHeroPos.y = 0.f;
581 }
582 // Checks for collision on bottom of map
583 if ((currentHero->getPos().y+currentHero->getFrameSize()+currentHero->getFrameSize()/2) > mapSize.y*16*4 ) {
584 currentHeroPos.y = mapSize.y*16*4 - currentHero->getFrameSize()-currentHero->getFrameSize()/2;
585 }

587 // View movement
588 currentHeroPosReal = sf::Vector2f(window.mapCoordsToPixel(currentHeroPos));
589 minViewSize = sf::Vector2f(window.mapCoordsToPixel(sf::Vector2f(0,0)));
590 maxViewSize = sf::Vector2f(window.mapCoordsToPixel(sf::Vector2f(mapSize.x*16*4,mapSize.y*16*4)));
591 sf::Vector2f viewMoveSpeed = sf::Vector2f(0.f, 0.f);
592 if (((currentHeroPosReal.x+16/2*4-window.getSize().x/2) < 0) && (minViewSize.x < 0) && currentHeroVelocity.x < 0) {
593 viewMoveSpeed.x = (currentHeroPos.x-view.getCenter().x)*time;
594 }
595 if (((currentHeroPosReal.x+16/2*4+window.getSize().x/2) > window.getSize().x) && (maxViewSize.x > window.getSize().x) && currentHeroVelocity.x > 0) {
596 viewMoveSpeed.x = ((currentHeroPos.x+16/2*4*2)-view.getCenter().x)*time;
597 }
598 if (((currentHeroPosReal.y+16/2*4-window.getSize().y/2) < 0) && (minViewSize.y < 0) && currentHeroVelocity.y < 0) {
599 viewMoveSpeed.y = (currentHeroPos.y-view.getCenter().y)*time;
