134 void Hero::walk(float const& dt, const float dir_x, const float dir_y, int frameRate) {
135 float currentX = getPos().x;
136 float currentY = getPos().y;
137 float currentSpeed = speed;
138
139 // Normalized diagonal speed. Since only diagonal movement is possible besides horizontal or vertical,
140 // sine of pi/4 (45*) is used for both directions as both cosine and sine of 45* have the same value.
141 if ((dir_x != 0) && (dir_y != 0)) {
142 currentSpeed *= sin(M_PI/4);
143 }
144
145 setPos(currentX+=(currentSpeed*dir_x*dt), currentY+=(currentSpeed*dir_y*dt));
146
147 if ((dir_x != 0) || (dir_y != 0)) {
148 currentState=Move;
149 }
150
151 if ((dir_x < 0) && (flipped == false)) {
152 flipped = true;
153 loadTexture(frameRate, flipped);
154 }
155 else if ((dir_x > 0) && (flipped == true)) {
156 flipped = false;
157 loadTexture(frameRate, flipped);
158 }
